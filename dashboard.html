# Lettura file Excel
dati <- read_excel("consumo_farmaci.xlsx")

# Pulizia colonna spesa_totale
dati <- dati %>%
  mutate(spesa_totale = readr::parse_number(as.character(spesa_totale)))

Tabella interattiva

datatable(dati, options = list(pageLength = 10))

Spesa totale per categoria ATC

ggplot(dati, aes(x = atc1, y = spesa_totale, fill = atc1)) +
  geom_boxplot() +
  labs(title = "Distribuzione della spesa per categoria ATC",
       x = "Categoria ATC",
       y = "Spesa (€)") +
  theme_minimal()

Spesa media per descrizione ATC

dati %>%
  group_by(descrizione_atc1) %>%
  summarise(spesa_media = mean(spesa_totale, na.rm = TRUE)) %>%
  ggplot(aes(x = reorder(descrizione_atc1, spesa_media), y = spesa_media)) +
  geom_col(fill = "steelblue") +
  coord_flip() +
  labs(title = "Spesa media per classe terapeutica",
       x = "Classe ATC",
       y = "Spesa media (€)") +
  theme_minimal()

Confronto Pre/Post COVID

dati %>%
  mutate(periodo = if_else(anno < 2020, "Pre-COVID", "Post-COVID")) %>%
  group_by(periodo, atc1) %>%
  summarise(spesa_media = mean(spesa_totale, na.rm = TRUE)) %>%
  ggplot(aes(x = atc1, y = spesa_media, fill = periodo)) +
  geom_col(position = "dodge") +
  labs(title = "Confronto spesa media Pre vs Post COVID per categoria ATC",
       x = "Categoria ATC",
       y = "Spesa media (€)") +
  theme_minimal()
`summarise()` has grouped output by 'periodo'. You can override using the
`.groups` argument.

dati %>%

  summarise(spesa_totale = sum(spesa_totale, na.rm = TRUE)) %>%
  ggplot(aes(x = reorder(regione, spesa_totale), y = spesa_totale)) +

  coord_flip() +

Previsione con Prophet

dati_prophet <- dati %>%
  group_by(anno) %>%
  summarise(spesa = sum(spesa_totale, na.rm = TRUE)) %>%
  mutate(ds = as.Date(paste0(anno, "-01-01")),
         y = spesa) %>%
  select(ds, y)

modello <- prophet(dati_prophet)
